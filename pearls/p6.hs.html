partitions :: [Digit] -> [[[Digit]]]

 111 
partitions [] = [[]]

 112

extend                     ::    Digit -> [Expression] -> [Expression]

 119 
extend x []             =    [[[[x]]]]

 120 
extend x es             =    concatMap (glue x) es

glue                     ::    Digit -> Expression -> [Expression]

 126
